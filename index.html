function haveSameData(obj1, obj2) {
	//Функція вертає true якщо два об'єкта однакові, і false якщо ні
	const obj1Length = Object.keys(obj1).length;
	const obj2Length = Object.keys(obj2).length;
	//Обчислюємо кількість ключей в обєктах
							
		if (obj1Length === obj2Length) {
			//далі задопомогою методу every перевіряємо
			//чи підходить кожен елемент масива під умову
			return Object.keys(obj1).every(
				key => obj2.hasOwnProperty(key)
					&& obj2[key] === obj1[key]);
		}
		return false;
}

    function* chunkArrayGenerator(array, size) {
	//Повертає чанки з певного масиву за розміром
	for (let i = 0; i < array.length; i += size) {
		//Через цикл проходимо по масиву вираховуємо
		//індекси кожного чанку і повертаємо зріз чанку
		yield array.slice(i, i + size);
	}
}

    function bulkRun(data) {
	let cbs = data.map(([f, arguments]) => {
	  return new Promise((resolve) => {
		f(...arguments, (result) => {
			resolve(result);
		 });
	  });
	});
 
	return Promise.all(cbs);
 }
 
 const f1 = (cb) => {
	cb(1);
 };
 
 const f2 = (a, cb) => {
	cb(a);
 };
 
 const f3 = (a, b, cb) => {
	setTimeout(() => cb([a, b]), 1000);
 };
 
 bulkRun([
	[f1, []],
	[f2, [2]],
	[f3, [3, 4]],
 ]).then(console.log);

    function arrayToObject(array) {
	//Перетворює масив в об'єкт
  const obj = {};
  for (const [key, value] of Object.entries(array)) {
	//Записуємо ключ та значення перетворенного масива
	  if (key.length > 0) {
		  if (Array.isArray(value)) {
			//Рекурсивно викликаємо функцію та перетворюємо 
			//вкладений масив в об'єкт, поки це масив
			  obj[key] = arrayToObject(value); // Рекурсія
		  } else {
			//Якщо це не масив, записуємо значення в об'єкт
			  obj[key] = value;
		  }
	  }
  }
  
  return obj;
  
}

    function objectToArray(obj) {
	//Перетворюємо об'єкт в масив
  const arr = [];

  for (const [key, value] of Object.entries(obj)) {
	  if (typeof value === 'object' && !Array.isArray(value)) {
		  //Рекурсивно викликаємо функцію для перетворення вкладених об'єктів в масив
		  arr.push([key, objectToArray(value)]); // Рекурсія для вкладених об'єктів
	  } else {
		//Записуємо в масив значення та ключ
		  arr.push([key, value]);
	  }
  }

  return arr;
}
const object = {
  name: 'developer',
  age: 5,
  skills: {
	  html: 4,
	  css: 5,
	  js: 5
  }
}
console.log(objectToArray(object));

    function NotificationException() { }
function ErrorException() { }
function primitiveMultiply(a, b) {
	const rand = Math.random();
	if (rand < 0.5) {
		return a * b;
	} else if (rand > 0.85) {
		throw new ErrorException()
	} else {
		throw new NotificationException()
	}
}

function reliableMultiply(a, b) {
	try {
		return primitiveMultiply(a, b)
	}
	catch (e) {
		if (e instanceof NotificationException) {
			//Через instanceof перевіряємо чи є об'єкт (e)
			//екземпляром функції (NotificationException) -> пробуємо знову отримати число
			return reliableMultiply(a, b)
		} else if (e instanceof ErrorException) {
			//Через instanceof перевіряємо чи є об'єкт (e)
			// екземпляром функції (ErrorException) -> завершуємо роботу
			return
		}
	}
}

    function separationObj(obj, parentKey = '') {
	//Об'єкт перетворюємо в плоску карту
	let result = {};

	for (const key in obj) {
		if (obj.hasOwnProperty(key)) {
			//Створюємо ключ карти в залежності від батьківського ключа
			const currentKey = parentKey ? `${parentKey}/${key}` : key;

			if (obj[key] && typeof obj[key] === 'object' && !Array.isArray(obj[key])) {
				//Рекурсивно викликаємо функцію і записуємо в result, поки значення ключа це об'єкт
				Object.assign(result, separationObj(obj[key], currentKey));
			} else {
				//В згенерований шлях карти (ключ) записуємо значення цього шляху
				result[currentKey] = obj[key];
			}
		}
	}
	return result;

}
const obj = {
	a: {
		b: {
			c: 12,
			d: 'Hello World'
		},
		e: [1, 2, 3]
	}
};

console.log(separationObj(obj));

    function combos(num) {
	//Створюємо комбінації чисел для досягнення певної суми
	const results = [];
  
	function generate(targetSum, currentCombo, startNum) {
	  if (targetSum === 0) {
		//Добавляємо результат першої згенерованої суми
		 results.push([...currentCombo]);
		 return;
	  }
  
	  for (let i = startNum; i <= num; i++) {
		 if (targetSum - i >= 0) {
			//Додаємо значення поки поточна комбінація не досягне суми
			currentCombo.push(i);
			generate(targetSum - i, currentCombo, i);
			currentCombo.pop();
		 } else {
			//Закінчимо генерацію першої суми
			break;
		 }
	  }
	}
  
	generate(num, [], 1);
  
	return results;
 }
  
 console.log(combos(10));

     function add(x) {
	//Сумує нескінченну кількість аргументів
	const sum = (y) => add(x + y);
	sum.valueOf = () => x;
	//Повертаємо стрілочну функцію яка рекурсивно сумує числа
	return sum;
 }
 
 //При виклиці Number(add(1)(2)(5)(-3)),
 //рекурсивно рахується сума всіх чисел і повертається число (сума)
 Number.prototype.add = function (x) {
	return add(this + x);
 };
